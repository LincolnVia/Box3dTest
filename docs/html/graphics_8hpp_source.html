69inline Matrix B3WorldTransformToMatrix(const b3WorldTransform& t, b3Vec3 scale)
70{
71 Matrix S = MatrixScale(scale.x, scale.y, scale.z);
72 Matrix R = QuaternionToMatrix(B3QuatToQuaternion(t.q));
